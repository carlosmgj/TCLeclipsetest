proc g {gk val} {
    puts "[ expr $val*1 ]"
    if { $gk == 1 } { set constantes [list 2 1/4 1/8] } \
    elseif { $gk == 2 } { set constantes [list -(1/64) -1/2 1/64] }
    set ret 0
    foreach op $constantes {
        if { $op < 0 } { 
            puts "Negative"
            puts "op a= $op"
            set  op [expr $op]
            puts "op b= $op"
            puts "val= $val"
            puts "ret= $ret"
            set ret [expr $ret-$op*$val]
            puts "$ret"} \
        elseif { $op > 0 } {
            puts "Positive"
            puts "op= $op"
            puts "val= $val"
            puts "ret= $ret"
            puts [expr $val*$op]
            set ret [expr $val*$op] 
            puts "$ret"
        }
        puts "--> $ret"
    }
    puts "$ret"
}
